program {
	factorial(int n)-> int{
		int res = 1;
		int i = 1;
		while (i <= n) {
			res = res * i;
            i = i + 1;
        }
		return res;
		}
	}
	main()-> void{
		print("Result = ", @factorial(5));
	}
}
